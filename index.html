# Core agent — modular, typed, testable
class PriceMonitorAgent:
    def __init__(self, api_client: Optional[PriceAPIClient] = None):
        self.api_client = api_client or PriceAPIClient()
        self.history: list[PriceRecord] = []

    def fetch_price(self, product_id: str) -> PriceRecord:
        # 1. Sanitize input — prevent injection
        safe_id = sanitize_product_id(product_id)
        if not safe_id:
            raise ValueError(f"Invalid product_id: {product_id!r}")

        # 2. Call external API
        raw = self.api_client.get_price(safe_id)

        # 3. Validate & normalize price
        price = sanitize_price(raw.get("price"))
        return PriceRecord(product_id=safe_id, current_price=price, ...)

    def evaluate_alerts(self, record: PriceRecord) -> bool:
        config = self._alert_configs.get(record.product_id)
        threshold = config.target_price * (1 + config.tolerance_pct)
        if record.current_price <= threshold:
            send_alert(to=config.alert_email, ...)  # 📧 fire!
            return True
        return False

jobs:
  lint:
    steps: [ruff check, black --check, mypy]

  tests:
    needs: lint
    steps: [pytest --cov=src --cov-fail-under=80]

  security:
    needs: lint
    steps: [snyk test --severity-threshold=high,
            snyk code test]

  build:
    needs: [tests, security]
    if: github.ref == 'refs/heads/develop'
    steps: [docker build-push :staging, :$SHA]

  deploy-staging:
    environment: staging
    steps: [SSH → docker pull && docker run]
